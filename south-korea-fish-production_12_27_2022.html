10. Rolling Forecast Origin to improve prediction capability¶

reference1: https://github.com/ritvikmath/Time-Series-Analysis/blob/master/SARIMA%20Model.ipynb¶

reference 2: https://parklize.blogspot.com/2019/06/how-to-forecase-time-series-in-python.html¶

def rolling_prediction(a,b,c):
    rolling_predictions = test_data.copy()
    predictions = list()
    for train_end in test_data.index:
        train_data = df_SK[:train_end-timedelta(days=365)]
        model = ARIMA(train_data, order=(a,b,c))
        model_fit = model.fit()

        pred = model_fit.forecast()
        yhat = pred[0]
        predictions.append(yhat)
        rolling_predictions[train_end] = pred
    mae = mean_absolute_error(test_data["Fish Production"], predictions)
    mse = mean_squared_error(test_data["Fish Production"], predictions)
    return predictions, mae, mse
